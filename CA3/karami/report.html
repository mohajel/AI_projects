class Bayes:
    def __init__(self): 
        self.normalizer = Normalizer()
        self.lemmatizer = Lemmatizer()
        self.stop_words = open("ignore.txt").read().split()
        
    def preprocess_text(self, text):
        text = ''.join([i for i in text if not i.isdigit()]) #delete numbers
        text = self.normalizer.normalize(text) #using semi space
        tokens = word_tokenize(text)
        chosen_tokens = []
        for token in tokens:
            if token not in self.stop_words:
                chosen_tokens.append(self.lemmatizer.lemmatize(token).split('#')[0])
        return chosen_tokens
فاز دوم: فرآیند مسئله

۲. $$ P(c|x) = \frac{P(x|c)P(c)}{P(x)} $$
x: مجموعه مرتب کلمات دیده شده.
c: موضوع مقاله

$P(c | x)$ : (Posterior Probability) احتمال اینکه موضوع خبر c باشد به شرط اینکه x در آن دیده شود.
روش محاسبه: این احتمال با استفاده از سه متغیر دیگر محاسبه می‌شود.

$P(x | c)$: (Likelihood) احتمال اینکه x دیده شود، به شرط اینکه موضوع مقاله c باشد.
روش محاسبه: به ازای هر مجموعه مرتب کلمات مانند y که در مقاله‌هایی با موضوع c آمده، محاسبه می‌کنیم که y چند بار در این مقاله ها تکرار شده و این مقدار را برابر با f(y) قرار می‌دهیم. سپس جمع تمام f(y) ها را به ازای هر y برابر s(c) در نظر می‌گیریم و در نهایت داریم: $$P(x | c) = \frac{f(x)}{s(c)}$$ :
$P(c)$: (Class Prior Probablity) احتمال اینکه موضوع مقاله با توجه به مشاهداتی که تا الان انجام شده است، c باشد.
روش محاسبه: این مقدار به مرور زمان عوض میشود و در هر مرحله برابر $P(x | c)$ می‌شود ولی در ابتدا محاسبه می‌کنیم که چند مقاله با موضوع c داریم و این مقدار را r(c) می‌نامیم. تعداد کل مقالات را نیز n می‌نامیم. و در نهایت داریم: $$P(c) = \frac{r(c)}{n}$$ حال در اینجا چون از تمام موضوع ها به یک اندازه مقاله داریم، P(c) به ازای تمام موضوعات برابر 0.25 است.

$P(x)$: (Predictor Prior Probablity) احتمال اینکه x دیده شود.
روش محاسبه: ابتدا به ازای هر مجموعه کلمات مانند y به دست می‌آوریم که در تمام مقالات چند بار تکرار شده اند و سپس این مقدار را g(y) می‌نامیم. سپس جمع تمام g(y) ها به ازای هر y را t می‌نامیم و در نهایت داریم: $$P(x) = \frac{g(x)}{t}$$

توجه کنید در اینجا x می‌تواند یک کلمه و یا مجموعه ای مرتب از کلمات باشند که پشت سر هم آمده‌اند مانند (bigrams).

def fix_titles(self, data):
    self.titles = dict()
    self.papers = len(data)
    for index, row in data.iterrows():
        if row.isnull().values.any():
            continue

        if row[1] in self.titles:
            self.titles[row[1]] += 1
        else :
            self.titles[row[1]] = 1

def fix_words(self, data):
    self.words = dict()
    self.total_words = 0
    
    self.words_in_title = dict()
    self.total_words_in_title = dict()
    self.distinct_words = 0
    seen_word = dict()
    for key, value in self.titles.items():
        self.words_in_title[key] = dict()
        self.total_words_in_title[key] = 0
        
    for index, row in data.iterrows():
        if row.isnull().values.any():
            continue
        title = row[1]
        tokens = self.preprocess_text(row[0])
        for token in tokens:
            self.total_words_in_title[title] += 1
            self.total_words += 1

            if token not in seen_word:
                self.distinct_words += 1
                seen_word[token] = True
            
            if token in self.words:
                self.words[token] += 1
            else:
                self.words[token] = 1

            if token in self.words_in_title[title]:
                self.words_in_title[title][token] += 1
            else:
                self.words_in_title[title][token] = 1
                

def train(self, data):
    self.fix_titles(data)
    self.fix_words(data)

Bayes.fix_titles = fix_titles
Bayes.fix_words = fix_words
Bayes.train = train

Additive Smoothing

۴. با توجه به فرمول $$ P(c|x) = \frac{P(x|c)P(c)}{P(x)} $$ در صورتی که x در c دیده نشود $P(x | c)$ برابر 0 می‌شود و در نتیجه $P(c|x)$ نیز برابر 0 می‌شود. پس مدل با قطعیت تشخیص می‌دهد که موضوع مورد نظر c نیست.
پس اگر x فقط در یک موضوع خاص مانند j وجود داشته باشد، مدل با قاطعیت تشخیص می‌دهد هر مقاله ای که در آن کلمه x آمده، موضوعش j است چون $p(c | x)$ به ازای تمام موضوعات دیگر برابر 0 است.

5. این روش روی محاسبه $P(x|c)$ تاثیر می‌گذارد، به صورتی که به جای محاسیه $\frac{f(x)}{s(c)}$ عبارت $\frac{f(x) + \alpha}{s(c) + unique\_words}$ را محاسبه می‌کنیم که $\alpha$ یک عدد ثابت صحیح و مثبت و $unique\_words$ تعداد کلمات متفاوتی است که در مقالات آمده‌اند. ( $s(c)$ برابر تعداد کل کلماتی است که در مقاله های با موضوع c آمده)
با این کار اگر x در مقالات با موضوع c نیامده باشند $P(x | c)$ دیگر برابر 0 نمی‌شود و در نتیجه $P(c | x)$ نیز 0 نمی‌شود و مدل ما با قاطعیت رد نمی‌کند که موضوع مقاله مورد نظر c نباشد.

def calculate_probs(self, text, alpha):
    tokens = self.preprocess_text(text);
    probs = dict()
    for title, value in self.titles.items():
        probs[title] = np.log(value / self.papers)

    for token in tokens:
        for title, value in self.titles.items():
            if token not in self.words_in_title[title]:
                self.words_in_title[title][token] = 0
            
            probs[title] += np.log((self.words_in_title[title][token] + alpha ) / (self.total_words_in_title[title] + ((alpha != 0) * self.distinct_words)))
    return probs
            
def predict_title(self, text, alpha):
    probs = self.calculate_probs(text, alpha)
    title = 'سلامت و زیبایی'
    max_prob = -float('inf')
    for key, value in probs.items():
        if value > max_prob:
            title = key
            max_prob = value
    return title

Bayes.calculate_probs = calculate_probs
Bayes.predict_title = predict_title
۷. Precision: در این معیار ما توجهی به تعداد مقاله‌هایی که این موضوع را داشته‌اند ولی موضوعشان اشتباه تشخیص داده شده است نداریم. مثلا فرض کنید ۱۰۰ مقاله علمی داریم و تنها یکی از این ۱۰۰ مقاله را علمی تشخیص دادیم و در کل هیج مقاله ی دیگری را علمی تشخیص نداده ایم، در این حالت مشخص است که ما دقت کمی داریم ولی این معیار برابر ۱ است که خیلی زیاد است.
به بیان دیگر به false negative ها اهمیت نمی‌دهیم.
Recall: مشکل این معیار دقیقا برعکس معیار قبلی است، در این معیار ما توجهی به مقاله‌هایی که این موضوع را نداشته اند و به اشتباه تشخیص داده ایم که این موضوع را دارند نمی‌شود. مثلا فرض کنید تشخیص دهیم موضوع تمام مقالات علمی است، واضح است که تشخیص ما دقت خیلی کمی دارد ولی مقدار این معیار برای این تشخیص 1 میشود که مقدار خیلی زیادی است.
به عبارت دیگر توجهی به false positive نداریم.

۸. این معیار از harmonic mean دو معیار بالا استفاده می‌کند و در نتیجه بر خلاف دو معیار بالا هم به false negative و هم به false positive توجه می‌کند.
فهم این معیار کمی از Accuracy سخت تر است ولی استفاده ی بیشتری دارد. به عبارت دیگر Accuracy زمانی بهتر کار می‌کند که false positive و false negative برای ما هزینه ی یکسانی داشته باشند، اگر هزینه این دو برای ما تفاوت زیادی داشته باشد، استفاده از F۱ score بهتر راست و برای همین این نوع میانگین گیری در اینجا اهمیت زیادی دارد.
۹. macro: در این روش از معیار F1-score در تمام کلاس ها میانگین می‌گیریم. به این معیار macro-average F1-score یه با اختصار macro-F1 نیز گفته می‌شود.
به طور مشابه macro-averaged F1-score و یا به اختصار macro-averaged تعریف می‌شود که در آن به جای معیار F1 برای کل کلاس ها از Presicion آن ها میانگین گیری می‌کنیم.

weightd: در معیار بالا ما برای میانگین گیری وزن یکسانی به F1-score تمام موضوعات می‌دهیم، اما در weighted-average F1-score یا به اختصار weightd-F1 به F1-score هر موضوع متناسب با تعداد دیتای موجود از هر موضوع وزن می‌دهیم و سپس میانگین گیری وزن دار انجام می‌دهیم.
به طور مشابه معیار های Weighted-precision و Weighted-recall نیز تعریف می‌شوند.

micro: این معیار در اصل برابر همان Accuracy است. به بیان دیگر در اینجا داریم:

$$micro-F1 = \frac{True Positives}{All\_test\_cases}$$

def calculate_accuracy(self, true_values, predicted_values):
    true_gues= 0;
    for i, j in zip(true_values, predicted_values):
        true_gues += int(i == j)
    return true_gues / len(true_values)
Bayes.calculate_accuracy = calculate_accuracy

def calculate_precision(self, true_values, predicted_values, label):
    correct_gues = 0
    all_gues = 0
    for i, j in zip(true_values, predicted_values):
        if j == label:
            all_gues += 1
            if i == j:
                correct_gues += 1
    return correct_gues / all_gues
Bayes.calculate_precision = calculate_precision

def calculate_recall(self, true_values, predicted_values, label):
    correct_gues = 0
    in_label = 0
    for i, j in zip(true_values, predicted_values):
        if i == label:
            in_label += 1
            if i == j:
                correct_gues += 1
    return correct_gues / in_label
Bayes.calculate_recall = calculate_recall

def calculate_f1(self, true_values, predicted_values, label):
    recall = self.calculate_recall(true_values, predicted_values, label)
    precision = self.calculate_precision(true_values, predicted_values, label)
    if (recall + precision == 0):
        return 0
    return 2 * (recall * precision) / (recall + precision)
Bayes.calculate_f1 = calculate_f1

def calculate_macro(self, true_values, predicted_values):
    f1 = []
    for label, value in self.titles.items():
        f1.append(self.calculate_f1(true_values, predicted_values, label))
    return sum(f1) / len(f1)
Bayes.calculate_macro = calculate_macro

def calculate_weighted(self, true_value, predicted_values):
    weights = dict()
    f1 = dict()
    
    for label, value in self.titles.items():
        weights[label] = 0
        f1[label] = (self.calculate_f1(true_values, predicted_values, label))
    for label in true_value:
        weights[label] += 1
        
    res = 0
    for label, value in f1.items():
        res += value * weights[label]
    return res / sum(weights.values())
Bayes.calculate_weighted = calculate_weighted

Touchline می‌تواند به کمک شما بیاید چرا که در این حالت بازی‌کنان کناری به گوشه و کنار زمین می‌روند و مرتبا بین زمین خود و رقیب جابه‌جا می‌شوند‌در این ترکیب مهم نیست که بازی مالکانه را ترجیح می‌دهید یا روی ضدحمله‌ها حساب می‌کنید‌ ارسال‌ پاس‌های بلند و هوایی و پراکندگی بازی‌کنان در گوشه و کنار زمین مهم‌ترین عامل موفقیت در حملات هستند‌با اینکه استفاده از ترکیب در بخش حمله حسابی به کمک شما می‌آید، اما از طرفی ممکن است با مشکلاتی در بخش دفاعی روبه‌رو شوید، چرا که بال‌های دفاعی شما دائما در حال رفت و آمد بین دو نیمه هستند و با کوچکترین غفلتی، جای‌شان خالی می‌شود‌ برای حل این مشکل پیشنهاد می‌کنیم حداقل پست دوتا از سه هافبک خود را به هافبک دفاعی (DMF) تغییر و با کمک آن‌ها ضعف‌های خط دفاعی را پوشش دهید‌بنابراین، در این ترکیب می‌بایست در پست بال‌ تهاجمی خود از بازی‌کنان تکنیکی و سرعتی استفاده کنید و در وسط آن‌ها، یک بازی‌کن هدف و تمام‌کننده قرار دهید‌ در خط میانی، استفاده از هافبک‌های دفاعی که استقامت و قدرت دفاعی بالایی دارند قطعا کمک‌کننده خواهد بود‌ترکیب ۱-۲-۱-۲-۴ (۱-۲-۳-۴ در بازی) اگر بازی‌کننده صبوری هستید و با حوصله بازی می‌کنید، این ترکیب انتخاب خوبی برای شما است‌هدف شما در ترکیب بالا این است که با پاس‌های کوتاه و حفظ مالکیت توپ، به سمت دروازه حریف حرکت کنید‌ اساسا برای انجام پاس‌های کوتاه و حفظ مالکیت باید سعی کنید تا همواره هم‌تیمی‌های خود را نزدیک به خود و در مسیر توپ-ن-ه دارید‌ در بخش تنظیمات تاکتیکی، بازی گزینه‌های زیادی را مقابل‌تان می‌گذارد‌در قسمت حالت حمله، حتما از گزینه بازی مالکانه (Possessive Attacking Style) استفاده کنید و سعی کنید با استفاده از پاس‌های کوتاه در وسط زمین، به سمت دروازه حریف حرکت کنید‌ همچنین سعی کنید تا بازی‌کنان خود را همواره در پست‌های خود نگه دارید و مطمئن شوید که همیشه سر جای خودشان باقی مانده‌اند‌ استفاده از گزینه تیکی-تاکا (Tiki-Taka) در بخش تنظیمات ترکیب باعث می‌شود تا انجام این کار ساده‌تر شود‌در بخش دفاعی از آنجایی که همیشه گروهی از بازی‌کنان نزدیک بهم هستند، خیلی مهم است که بعد از لو دادن توپ و پیش از اینکه حریف فضاهای خالی را پیدا کند، آن را پس بگیرید و به بازی مالکانه خود ادامه دهید‌ در این مواقع باید سعی کنید بازی‌کنان حریف را تا جای ممکن پرس کنید و آن‌ها را تحت فشار قرار دهید‌ به همین منظور، در بخش تنظیمات ترکیب حالت‌های دفاعی خود را حتما روی Aggressive Pressuring و Frontline Pressure قرار دهید تا رقیب‌تان برای ادامه کار به دردسر بیوفتد‌در سمتی دیگر استفاده از دو هافبک دفاعی (DMF) قدرتمند می‌تواند خط دفاعی تیم‌تان را تقویت کند‌ اگر حریف‌تان از ضدحمله‌های سریع و برق‌آسا استفاده می‌کند، پیشنهاد می‌کنیم که حتما از دو هافبک دفاعی استفاده کنید‌در ترکیب بالا ضمن اینکه باید قدرت پاسکاری و دریبل‌زنی بالایی داشته باشید، باید حسابی صبور باشید و با حوصله پیش بروید‌ترکیب ۲-۳-۵ (۲-۱-۲-۵ در بازی) اگر اهمیت زیادی برای بخش دفاعی تیم‌تان قائل هستید، استفاده از این ترکیب را برای‌تان پیشنهاد می‌کنیم‌با اینکه تمرکز ترکیب بالا روی دفاع کردن است، اما استفاده از دو مهاجم وسط می‌تواند حسابی خط دفاعی تیم حریف را به دردسر بیندازد و اثر ویرانگری داشته باشد‌ در فاز حمله، کلید موفقیت این است که توپ را در سریع‌ترین زمان ممکن به یکی از دو مهاجم وسط خود برسانید‌ بنابراین در بخش تنظیمات ترکیب، از حالت ضدحمله (Counter Attacking Style) استفاده کنید و مرتب از پاس‌های هوایی و بلند بهره ببرید. اگر می‌خواهید شجاعت بیش‌تری به خرج دهید، می‌توانید مدافعین کناری خود را ماموریت کنید تا در زمان حمله به سمت دروازه حریف بیایند و مشارکت کنند. در بخش دفاع لازم نیست کار خیلی خاصی انجام دهید چون این ترکیب اساسا یک ترکیب دفاعی است و بیشتر بازی‌کنان در کار دفاعی تیم شرکت می‌کنند. از طرفی آنقدر فضای خط دفاعی تیم فشرده است که حریف‌تان به سختی می‌تواند کاری از پیش ببرد. اگر می‌خواهید از این ترکیب استفاده کنید، اول از هر چیز به سه مدافع میانی قدرتمند نیاز دارید تا کارهای خط دفاعی را به بهترین شکل ممکن انجام دهند. علاوه بر این، استفاده از یک مهاجم نوک قدرتمند و تمام‌کننده در خط حمله بسیار مهم است. ترکیب ۳-۱-۲-۴اگر هر سال به صورت مرتب فوتبال بازی می‌کنید و جزء دسته حرفه‌ای‌ها هستید، استفاده از این ترکیب را برای‌تان پیشنهاد می‌کنیم. جالب است بدانید که قهرمان حال حاضر بازی – Usmakabyle – هم از این ترکیب استفاده می‌کند. در ترکیب بالا، حمله حرف اول و آخر را می‌زند. در واقع این ترکیب برای بازی‌کنان هجومی طراحی شده است و کسانی که از آن استفاده می‌کنند، باید با استفاده از هافبک‌های خود توپ را به دروازه برسانند. حرکت و چرخش سریع توپ بین نقاط مختلف زمین بسیار حیاتی است و باید تا جای ممکن از خط هافبک خود کار بکشید. کوتاه کردن فاصله بین بازی‌کنان و فشرده کردن آن‌ها به پاسکاری تیمی‌تان کمک بسیار زیادی خواهد کرد. در فاز دفاعی از آنجایی که از یک ترکیب هجومی استفاده می‌کنیم باید سعی کنید با پرس زیاد بازی‌کنان حریف، توپ را در اولین فرصت پس بگیرید. همچنین جلو کشیدن خط دفاعی فضای تیم حریف را محدود و راه آن‌ها را برای رسیدن به دروازه سخت می‌کند. با این حال، در استفاده از این استراتژی باید حسابی مراقب باشد و اگر کوچکترین اشتباهی کنید، احتمالا خودتان بیشتر از حریف از این موضوع ضربه خواهید خورد. همان‌طور که بالاتر گفتیم، در این ترکیب باید بازی‌کنان حریف را پرس کنید و در فاز حمله با سرعت زیادی توپ را به حرکت در بیاورید. بنابراین استقامت و قدرت بازی‌کنان تیم‌تان اهمیت بسیار زیادی در این بین دارد. از طرفی برای دوام آوردن تا دقیقه ۹۰ باید چند تعویض انجام دهید و بازی‌کنان تازه نفسی را به میدان بفرستید. استفاده از هافبک‌هایی که قدرت پاسکاری بالایی دارند و مهاجم‌های تمام‌کننده در این ترکیب بسیار مفید خواهد بود. منبع: GamesRadar

به عنوان مثال در نمونه دوم در بالا، درباره ی شرکت های بازی سازی و تکنولوژی آن ها صحبت شده و این موضوع این مقاله تکنولوژی است ولی چون فقط وجود کلمات برای ما مهم است و بیشتر کلمات این متن مانند AMD, گیمر، بازی و ... به بازی مربوط هستند و زیاد تکرار شده اند، مدل ما این مقاله را در دسته ی بازی ها دسته بندی می‌کند.
